Server side import socket

ip='localhost' #our ip

sock = socket.socket()
sock.bind((ip, 9091))
sock.listen(10)
conn, addr = sock.accept()

while True:
data = conn.recv(1024)
if not data:
break
if data.decode('utf-8').__contains__('/exit'):
break
print(data.decode('utf-8'))
conn.send(bytes(input("You: "), 'utf-8'))

conn.close()
